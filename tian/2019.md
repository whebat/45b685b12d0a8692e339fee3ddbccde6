# Part B

## Q1

```cpp
class Cat: public Animal {
    private:
    int lengthTail;
    int mass;
    int countTeeth;
    public:
    Cat() = default;
    Cat(int lt, int m, int ct, string name, int age)
    : Animal(name, age)
    {
        lengthTail = lt;
        mass = m;
        countTeeth = ct;
    }
    ostream operator<<(ostream &os)
    {
        os << "Tail: " << lengthTail << '\n';
        os << "Mass: " << mass << '\n';
        os << "Number of teeth: " << countTeeth << '\n';
        return os;
    }
};
```
